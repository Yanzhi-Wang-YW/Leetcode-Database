```py
import pandas as pd

def list_products(products: pd.DataFrame, orders: pd.DataFrame) -> pd.DataFrame:
    filter_date = orders.copy() 
    filter_date['order_date'] = pd.to_datetime(filter_date['order_date']) 
    filter_date = filter_date.loc[filter_date['order_date'].dt.strftime('%Y-%m') == '2020-02'] 
    customTransformations1 = filter_date
    count_unit = customTransformations1.copy() 
    count_unit = count_unit.groupby('product_id').agg(
    unit=('unit', 'sum')
    ).reset_index() 
    customTransformations2 = count_unit 
    filter_unit = customTransformations2.copy() 
    filter_unit = filter_unit.loc[filter_unit['unit'] >= 100] 
    customTransformations3 = filter_unit 
    join1 = pd.merge(customTransformations3, products, left_on=["product_id"], right_on=["product_id"], how="inner")
    select_column = join1.copy() 
    select_column = select_column[['product_name', 'unit']] 
    customTransformations4 = select_column 
    return customTransformations4     
```
