```py
import pandas as pd

def top_travellers(users: pd.DataFrame, rides: pd.DataFrame) -> pd.DataFrame:
    rides = rides.groupby('user_id').agg(
        travelled_distance=('distance', 'sum') 
    ).reset_index() 
    output = users.merge(rides, how='left', left_on='id', right_on='user_id') 
    output = output[['name', 'travelled_distance']] 
    output = output.sort_values(['travelled_distance', 'name'], ascending=[False, True]) 
    output = output.fillna(value=0)      
    return output 
```
