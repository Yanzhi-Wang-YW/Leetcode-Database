<p>Each node in the tree can be one of three types:</p>

<ul>
	<li><strong>"Leaf"</strong>: if the node is a leaf node.</li>
	<li><strong>"Root"</strong>: if the node is the root of the tree.</li>
	<li><strong>"Inner"</strong>: If the node is neither a leaf node nor a root node.</li>
</ul>

<p>Write a solution to report the type of each node in the tree.</p>

<p>Return the result table in <strong>any order</strong>.</p>

<p>The result format is in the following example.</p>

### join 
```py
import pandas as pd

def tree_node(tree: pd.DataFrame) -> pd.DataFrame: 
    root_node = tree.loc[tree['p_id'].isna()] 
    root_node = root_node[['id']] 
    root_node['type'] = 'Root' 
    inner_node = tree.loc[tree['p_id'].notna()]
    inner_node = inner_node.merge(inner_node, how='inner', left_on='id', right_on='p_id')   
    inner_node = inner_node[['id_x']] 
    inner_node = inner_node.drop_duplicates(subset=['id_x'])  
    inner_node = inner_node.rename(columns={'id_x': 'id'}) 
    inner_node['type'] = 'Inner' 
    leaf_node = tree.loc[tree['p_id'].notna()] 
    leaf_node = leaf_node.merge(leaf_node, how='left', left_on='id', right_on='p_id', indicator=True) 
    leaf_node = leaf_node.loc[leaf_node['_merge'] == 'left_only'] 
    leaf_node = leaf_node[['id_x']] 
    leaf_node = leaf_node.drop_duplicates(subset=['id_x']) 
    leaf_node = leaf_node.rename(columns={'id_x': 'id'}) 
    leaf_node['type'] = 'Leaf' 
    output = pd.concat([root_node, inner_node, leaf_node])          
    return output
```

### apply  
```py
import pandas as pd

def tree_node(tree: pd.DataFrame) -> pd.DataFrame:
    def classify(row):
        if pd.isna(row['p_id']):
            return 'Root'
        elif row['id'] in set(tree['p_id'].dropna().astype(int)):
            return 'Inner'
        else:
            return 'Leaf'
    
    tree['type'] = tree.apply(classify, axis=1)
    return tree[['id', 'type']]
```

### ny.where 
```py
import pandas as pd
import numpy as np

def tree_node(tree: pd.DataFrame) -> pd.DataFrame:
    tree['type'] = np.where(
        tree['p_id'].isna(),                     
        'Root',
        np.where(
            tree['id'].isin(set(tree['p_id'].dropna().astype(int))),                           
            'Inner',
            'Leaf'                               
        )
    )
    
    return tree[['id', 'type']]
```

### np.select
```py
def tree_node(tree: pd.DataFrame) -> pd.DataFrame:
    conditions = [
        tree['p_id'].isna(),               
        tree['id'].isin(set(tree['p_id'].dropna().astype(int)))        
    ]
    choices = ['Root', 'Inner']
    
    tree['type'] = np.select(conditions, choices, default='Leaf')
    return tree[['id', 'type']]
```
