```md
| quantity    | int     |

inventory_id is the unique identifier for this table.
Each row represents the inventory of a specific product at a specific store.
Write a solution to find stores that have inventory imbalance - stores where the most expensive product has lower stock than the cheapest product.

For each store, identify the most expensive product (highest price) and its quantity
For each store, identify the cheapest product (lowest price) and its quantity
A store has inventory imbalance if the most expensive product's quantity is less than the cheapest product's quantity
Calculate the imbalance ratio as (cheapest_quantity / most_expensive_quantity)
Round the imbalance ratio to 2 decimal places
Only include stores that have at least 3 different products
Return the result table ordered by imbalance ratio in descending order, then by store name in ascending order.

The result format is in the following example.
```

```py
import pandas as pd

def find_inventory_imbalance(stores: pd.DataFrame, inventory: pd.DataFrame) -> pd.DataFrame:
    # For each store, identify the most expensive product (highest price) and its quantity 
    most_expensive_product = inventory[['store_id', 'product_name', 'quantity', 'price']].copy()
    most_expensive_product = most_expensive_product.sort_values(['store_id', 'price'], ascending=[True, False]) 
    most_expensive_product['row_num'] = most_expensive_product.groupby('store_id').cumcount() + 1 
    most_expensive_product = most_expensive_product.loc[most_expensive_product['row_num'] == 1, ['store_id', 'product_name', 'quantity', 'price']] 
    most_expensive_product = most_expensive_product.rename(columns={'product_name':'most_exp_product', 'quantity':'most_exp_quantity', 'price':'most_exp_price'}) 
    # For each store, identify the cheapest product (lowest price) and its quantity 
    most_cheapest_product = inventory[['store_id', 'product_name', 'quantity', 'price']].copy()
    most_cheapest_product = most_cheapest_product.sort_values(['store_id', 'price'], ascending=[True, True]) 
    most_cheapest_product['row_num'] = most_cheapest_product.groupby('store_id').cumcount() + 1 
    most_cheapest_product = most_cheapest_product.loc[most_cheapest_product['row_num'] == 1, ['store_id', 'product_name', 'quantity', 'price']] 
    most_cheapest_product = most_cheapest_product.rename(columns={'product_name':'cheapest_product', 'quantity':'cheapest_quantity', 'price':'cheapest_price'}) 
    # A store has inventory imbalance if the most expensive product's quantity is less than the cheapest product's quantity 
    inventory_imbalance = most_expensive_product.merge(most_cheapest_product, how='inner', on='store_id') 
    inventory_imbalance = inventory_imbalance.loc[inventory_imbalance['most_exp_quantity'] < inventory_imbalance['cheapest_quantity']] 
    # Calculate the imbalance ratio as (cheapest_quantity / most_expensive_quantity) 
    inventory_imbalance['imbalance_ratio'] = (inventory_imbalance['cheapest_quantity'] / inventory_imbalance['most_exp_quantity'] * 100 + 0.5).astype(int) / 100.0 
    inventory_imbalance = inventory_imbalance[['store_id', 'most_exp_product', 'cheapest_product', 'imbalance_ratio']] 
    # Only include stores that have at least 3 different products  
    total_products = inventory[['store_id', 'inventory_id']].copy() 
    total_products = total_products.groupby('store_id').agg(
        cnt=('inventory_id', 'count') 
    ).reset_index() 
    total_products = total_products.loc[total_products['cnt'] >= 3, ['store_id']] 
    # output 
    output = inventory_imbalance.merge(total_products, how='inner', on='store_id')
    output = stores.merge(output, how='inner', on='store_id')   
    output = output.sort_values(['imbalance_ratio', 'store_name'], ascending=[False, True])          
    return output
```
