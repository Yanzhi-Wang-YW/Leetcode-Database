import pandas as pd

def sales_analysis(product: pd.DataFrame, sales: pd.DataFrame) -> pd.DataFrame: 
    output = sales.groupby('seller_id').agg(
        sum_price=('price', 'sum')
    ).reset_index() 
    output['rank_num'] = output['sum_price'].rank(method='dense', ascending=False) 
    output = output.loc[output['rank_num'] == 1]  
    output = output[['seller_id']]      
    return output
